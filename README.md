<script>
// Generate the 12 clock face tick markers dynamically
const ticksContainer = document.getElementById("ticks");
for (let i = 0; i < 12; i++) {
    const tick = document.createElement("div");
    tick.className = "tick";
    tick.style.transform = `rotate(${i * 30}deg)`;
    ticksContainer.appendChild(tick);
}

function update(){
    const now = new Date();

    let h = now.getHours();
    const m = now.getMinutes();
    
    // Calculate 12-hour AM/PM formatting values
    const ampm = h >= 12 ? 'PM' : 'AM';
    const displayHour = h % 12 === 0 ? 12 : h % 12;

    const hourAngle = (h % 12) * 30 + m * 0.5;
    const minuteAngle = m * 6;

    document.getElementById("hour").style.transform = `rotate(${hourAngle}deg)`;
    document.getElementById("minute").style.transform = `rotate(${minuteAngle}deg)`;

    // Display Digital String matching image style "11:15 AM"
    document.getElementById("digital").textContent = 
        String(displayHour).padStart(2, "0") + ":" + 
        String(m).padStart(2, "0") + " " + ampm;

    // Display Custom Date matching style "Fri Jul 17"
    const days = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];
    const months = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];

    document.getElementById("date").textContent = 
        days[now.getDay()] + " " + 
        months[now.getMonth()] + " " + 
        now.getDate();
}

update();
setInterval(update, 1000);
</script>
